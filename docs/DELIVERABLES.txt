QUICK START
================================================================================

1. IMPORT COMPONENTS:
   import { AgentUIWrapper, type StepConfig } from './components/AgentUIWrapper';
   import { AgentStep } from './components/AgentStep';

2. FORMAT DATA:
   const steps: StepConfig[] = [
     {
       id: 'step-1',
       title: 'Analyze',
       description: 'Analyzing market',
       status: 'active',
       isThinking: true,
       subItems: [
         { id: 'sub-1', type: 'query', label: 'Searching...' },
       ],
     },
   ];

3. CONNECT TO AGENT:
   agent.on('step', (step) => updateSteps(step));
   agent.on('thinking', (text) => setThinkingOutput(text));

4. RENDER:
   <AgentUIWrapper
     taskDescription="Create ads"
     steps={steps}
     isThinking={true}
     liveThinkingOutput={output}
   />

See AGENT_UI_REFACTOR.md for complete integration guide.
